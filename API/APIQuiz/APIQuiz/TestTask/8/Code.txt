using System;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            string result = "(";

            for (int i = 0; i < 10; ++i)
            {
                result += args[i];
                if (i == 2)
                    result += ") ";
                if (i == 5)
                    result += "-";
            }

            Console.Write(result);
        }
    }
}
