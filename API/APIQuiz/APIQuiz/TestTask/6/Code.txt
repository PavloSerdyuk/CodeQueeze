using System;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            string result = "(";

            for (int i = 0; i < 10; ++i)
            {
                result += args[i];
                if (i == 3)
                    result += ") ";
                if (i == 6)
                    result += "-";
            }

            Console.Write(result);
        }
    }
}
