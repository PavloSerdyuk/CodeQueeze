using System;

namespace TestingTask
{
    class Program
    {
        static void Main(string[] args)
        {
            for(int i =0; i < args.Length-1; i++)
            {
                if(Convert.ToInt32(args[i]) > Convert.ToInt32(args[i+1]))
                {
                    Console.Write("false");
                    return;
                }
            }
            Console.Write("true");
        }
    }
}
